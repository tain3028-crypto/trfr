local function GetDB()
    local success, response = pcall(function()
        return request({
            Url = DatabaseURL,
            Method = "GET"
        })
    end)
    if success and response.StatusCode == 200 and response.Body ~= "null" then
        return HttpService:JSONDecode(response.Body)
    end
    return {}
end

local function SaveDB(db)
    pcall(function()
        request({
            Url = DatabaseURL,
            Method = "PUT",
            Headers = {["Content-Type"] = "application/json"},
            Body = HttpService:JSONEncode(db)
        })
    end)
end

local function FormatTime(seconds)
    if seconds == -1 then return "Vĩnh viễn" end
    if seconds <= 0 then return "Đã hết hạn" end
    local days = math.floor(seconds / 86400)
    local hours = math.floor((seconds % 86400) / 3600)
    local mins = math.floor((seconds % 3600) / 60)
    if days > 0 then return string.format("%d ngày, %d giờ", days, hours) end
    return string.format("%d phút, %d giây", mins, seconds % 60)
end

local MyTabID = tostring(math.random(100000, 999999))

local function CheckAuth()
    local inputKey = Config.MyKey
    local currentTime = os.time()
    local db = GetDB()

    -- 1. Check xem key có trong danh sách trắng không
    local keyDuration = Config.DanhSachKey[inputKey]
    if not keyDuration then 
        return false, "Key không tồn tại trong hệ thống!" 
    end

    -- 2. Lấy dữ liệu trên Firebase
    if not db[inputKey] then
        db[inputKey] = {StartTime = 0, IsExpired = false, ActiveTabs = {}}
    end
    local kData = db[inputKey]

    -- 3. Check xem đã bị khóa vĩnh viễn chưa
    if kData.IsExpired then 
        return false, "Key này đã hết hạn sử dụng trước đó!" 
    end

    -- 4. Kích hoạt lần đầu
    if kData.StartTime == 0 then
        kData.StartTime = currentTime
    end

    -- 5. Tính toán thời gian còn lại
    local timeLeft = -1
    if keyDuration ~= -1 then
        local timePassed = currentTime - kData.StartTime
        timeLeft = keyDuration - timePassed
        
        if timePassed >= keyDuration then
            kData.IsExpired = true
            db[inputKey] = kData
            SaveDB(db)
            return false, "Thời hạn key đã hết ("..FormatTime(keyDuration)..")!"
        end
    end

    -- 6. Quản lý số Tab (Concurrency)
    local updatedTabs = {}
    local onlineCount = 0
    for id, lastSeen in pairs(kData.ActiveTabs or {}) do
        if (currentTime - lastSeen) < 15 then
            updatedTabs[id] = lastSeen
            onlineCount = onlineCount + 1
        end
    end

    if not updatedTabs[MyTabID] then
        if onlineCount >= Config.MaxTabs then
            return false, "Key đã đạt giới hạn " .. Config.MaxTabs .. " tab!"
        end
        onlineCount = onlineCount + 1
    end

    updatedTabs[MyTabID] = currentTime
    kData.ActiveTabs = updatedTabs
    db[inputKey] = kData
    
    SaveDB(db)
    return true, onlineCount, timeLeft
end

-- ══════════════════════════════════════════
--  safeFly  (anti-stuck tích hợp)
-- ══════════════════════════════════════════
local function safeFly(targetPos)
    character = player.Character or player.CharacterAdded:Wait()
    rootPart  = character:WaitForChild("HumanoidRootPart")
    enableNoclip()

    local bv = Instance.new("BodyVelocity")
    bv.MaxForce = Vector3.new(math.huge, math.huge, math.huge)
    bv.Velocity = Vector3.new(0,0,0)
    bv.Parent   = rootPart

    local ok = true

    -- Pha 1: Bay lên nếu thấp
    if rootPart.Position.Y < 50 then
        local ascend   = Vector3.new(rootPart.Position.X, CONFIG.safeHeight, rootPart.Position.Z)
        local tweenUp  = TweenService:Create(rootPart,
            TweenInfo.new((rootPart.Position - ascend).Magnitude / CONFIG.speed, Enum.EasingStyle.Linear),
            {CFrame = CFrame.new(ascend)})

        local prev, timer = rootPart.Position, 0
        local conn = RunService.Heartbeat:Connect(function(dt)
            timer += dt
            if timer >= CONFIG.stuckCheckTime then
                if (rootPart.Position - prev).Magnitude < CONFIG.stuckThreshold then
                    ok = false; tweenUp:Cancel()
                end
                timer = 0; prev = rootPart.Position
            end
        end)
        tweenUp:Play(); tweenUp.Completed:Wait(); conn:Disconnect()
    end

    if not ok then
        log("⚠ Anti-stuck (bay lên)! Teleport về chỗ chờ.")
        disableNoclip(); bv:Destroy(); emergencyTeleport(POS_WAIT); return false
    end

    -- Pha 2: Bay đến đích
    local tweenMain = TweenService:Create(rootPart,
        TweenInfo.new((rootPart.Position - targetPos).Magnitude / CONFIG.speed, Enum.EasingStyle.Linear),
        {CFrame = CFrame.new(targetPos)})

    local prev, timer = rootPart.Position, 0
    local conn = RunService.Heartbeat:Connect(function(dt)
        timer += dt
        if timer >= CONFIG.stuckCheckTime then
            if (rootPart.Position - prev).Magnitude < CONFIG.stuckThreshold then
                ok = false; tweenMain:Cancel()
            end
            timer = 0; prev = rootPart.Position
        end
    end)
    tweenMain:Play(); tweenMain.Completed:Wait(); conn:Disconnect()

    disableNoclip(); bv:Destroy()

    if not ok then
        log("⚠ Anti-stuck (bay chính)! Teleport về chỗ chờ.")
        emergencyTeleport(POS_WAIT); return false
    end
    return true
end

local function normalizeFruitName(raw)
    return (raw:match("^(.-)%-") or raw):lower():gsub("%s+","")
end

local _fruitSet, _cacheTime, CACHE_TTL = nil, 0, 30

local function fetchInventoryFruits()
    local commF = getCommF()
    if not commF then log("❌ CommF_ không khả dụng!"); return nil end
    local ok, inv = pcall(function() return commF:InvokeServer("getInventory") end)
    if not ok or not inv then log("❌ getInventory thất bại: "..tostring(inv)); return nil end
    local set, list = {}, {}
    for _, item in pairs(inv) do
        if item.Type == "Blox Fruit" then
            set[normalizeFruitName(item.Name)] = true
            table.insert(list, item.Name)
        end
    end
    log(string.format("--- 📦 KHO (%d trái) ---", #list))
    for i,n in ipairs(list) do log(string.format("  [%d] %s", i, n)) end
    log("------------------------")
    return set
end

local function invalidateInventoryCache()
    _fruitSet = nil; _cacheTime = 0
end

local function hasFruitInInventory(fruitName)
    local now = tick()
    if _fruitSet == nil or (now - _cacheTime) > CACHE_TTL then
        _fruitSet = fetchInventoryFruits(); _cacheTime = now
    end
    if _fruitSet == nil then
        log("⚠ Không đọc được kho. Giả định role = RECV."); return false
    end
    local target = fruitName:lower():gsub("%s+","")
    local found  = _fruitSet[target] == true
    log(string.format("🔍 Check kho: '%s' → %s", fruitName, found and "CÓ ✅" or "KHÔNG ❌"))
    return found
end

local tradeScriptRunning = false

local function runTradeScript()
    --[[
        Vòng lặp nội bộ: thêm trái vào bàn trade rồi Accept.
        Thoát ra khi tradeScriptRunning = false (do standUpAndReturn).
    ]]
    tradeScriptRunning = true
    log("▶ [SUB-TRADE] Khởi động script ACC CÓ TRÁI...")

    local tradeFunc = getTradeFunction()
    if not tradeFunc then
        warn("[SUB-TRADE] Không tìm thấy TradeFunction! Bỏ qua sub-script.")
        tradeScriptRunning = false; return
    end

    local fruitInfo = getFruitInfo(_G.FruitToTrade)
    if not fruitInfo then
        warn("[SUB-TRADE] Không có ID cho trái: ".._G.FruitToTrade)
        tradeScriptRunning = false; return
    end

    while tradeScriptRunning do
        log(string.format("[SUB-TRADE] Thêm trái '%s' (id=%d) vào bàn...", fruitInfo.name, fruitInfo.id))

        local ok1, r1 = pcall(function()
            return tradeFunc:InvokeServer("addItem", fruitInfo.id, 1)
        end)

        if ok1 then
            log("[SUB-TRADE] Đã thêm trái. Chờ đồng bộ 10s...")
            task.wait(10)

            log("[SUB-TRADE] Gửi Accept...")
            local ok2, r2 = pcall(function()
                return tradeFunc:InvokeServer("accept")
            end)
            if ok2 then
                log("[SUB-TRADE] ✅ Accept thành công!")
            else
                warn("[SUB-TRADE] ❌ Accept thất bại: "..tostring(r2))
            end
        else
            warn("[SUB-TRADE] ❌ addItem thất bại: "..tostring(r1))
        end

        -- Ẩn Hub nếu có
        pcall(function()
            if CoreGui:FindFirstChild("Banana Hub") then
                CoreGui["Banana Hub"].Enabled = false
            end
        end)

        if tradeScriptRunning then task.wait(1) end
    end

    log("⏹ [SUB-TRADE] Dừng sub-script TRADE.")
end

    local minValue = enemyInfo.beli * 0.6
    local maxValue = math.floor(enemyInfo.beli / 0.6)
    log(string.format("[SUB-RECV] Đối phương ném: %s (%d Beli) | Vùng hợp lệ: %d – %d",
        enemyInfo.name, enemyInfo.beli, minValue, maxValue))

    while recvScriptRunning do
        -- Lấy inventory tươi mỗi vòng lặp recv
        local ok0, inv = pcall(function() return commF:InvokeServer("getInventory") end)
        if not ok0 or not inv then
            warn("[SUB-RECV] Không đọc được kho. Thử lại sau 2s...")
            task.wait(2); continue
        end

        -- Build danh sách trái của mình (có thể nhiều trái cùng loại)
        local myFruits = {}
        for _, item in pairs(inv) do
            if item.Type == "Blox Fruit" then
                local info = getFruitInfo(item.Name)
                if info then
                    local count = item.Count or 1
                    for _ = 1, count do table.insert(myFruits, info) end
                end
            end
        end
        -- Sắp xếp đắt → rẻ
        table.sort(myFruits, function(a, b) return a.beli > b.beli end)

        local totalValue, addedCount = 0, 0

        for _, fruit in ipairs(myFruits) do
            if not recvScriptRunning then break end
            if addedCount >= 4 then break end
            if totalValue >= minValue then break end
            -- Lọc vượt 40%
            if totalValue + fruit.beli > maxValue then
                log(string.format("  [SUB-RECV] Bỏ qua %s (tổng sẽ vượt %d)", fruit.name, maxValue))
                continue
            end

            log(string.format("  [SUB-RECV] Thêm: %s (%d Beli)...", fruit.name, fruit.beli))
            local ok1, _ = pcall(function()
                return tradeFunc:InvokeServer("addItem", fruit.id, 1)
            end)
            if ok1 then
                totalValue  += fruit.beli
                addedCount  += 1
                log(string.format("  → OK! Tổng bàn: %d Beli.", totalValue))
            else
                log(string.format("  → Thất bại (đã đầy). Bỏ qua."))
            end
            task.wait(0.5)
        end

        -- Xử lý Accept
        if totalValue < minValue then
            warn(string.format("[SUB-RECV] ⚠ Kho không đủ giá trị (%d < %d). Bỏ vòng này.", totalValue, minValue))
        else
            log(string.format("[SUB-RECV] ✅ Tổng: %d Beli hợp lệ. Chờ đồng bộ 10s...", totalValue))
            task.wait(10)
            local ok2, r2 = pcall(function() return tradeFunc:InvokeServer("accept") end)
            if ok2 then log("[SUB-RECV] ✅ Accept thành công!")
            else         warn("[SUB-RECV] ❌ Accept thất bại: "..tostring(r2)) end
        end

        -- Ẩn Hub
        pcall(function()
            if CoreGui:FindFirstChild("Banana Hub") then
                CoreGui["Banana Hub"].Enabled = false
            end
        end)

        if recvScriptRunning then task.wait(1) end
    end

    log("⏹ [SUB-RECV] Dừng sub-script RECV.")
end

-- ══════════════════════════════════════════
--  Spawn đúng sub-script theo role
-- ══════════════════════════════════════════
local function startSubScript()
    if role == "trade" then
        --  Acc có trái → chạy Trade, dừng Recv nếu đang chạy
        recvScriptRunning = false
        if not tradeScriptRunning then
            task.spawn(runTradeScript)
        end
    elseif role == "recv" then
        --  Acc không có trái → chạy Recv, dừng Trade nếu đang chạy
        tradeScriptRunning = false
        if not recvScriptRunning then
            task.spawn(runRecvScript)
        end
    end
end

local function stopSubScripts()
    tradeScriptRunning = false
    recvScriptRunning  = false
end

-- ══════════════════════════════════════════
--  standUpAndReturn  (reset sau timeout)
-- ══════════════════════════════════════════
local function standUpAndReturn()
    log("🔄 Đứng dậy & bay về chỗ chờ...")
    currentSeatIdx = nil
    seatStartTime  = nil

    -- Dừng cả 2 sub-script trong lúc về chỗ chờ
    stopSubScripts()

    pcall(function()
        humanoid = character:FindFirstChildOfClass("Humanoid")
        if humanoid then humanoid.Jump = true end
    end)

    task.wait(0.5)
    safeFly(POS_WAIT)
    task.wait(1)

    -- Re-check inventory (trái có thể đã được trade xong)
    invalidateInventoryCache()
    log("♻ Reset cache. Re-xác định role...")
    determineRole()

    -- Khởi động lại đúng sub-script theo role mới
    startSubScript()
end

-- ══════════════════════════════════════════
--  Seat Logic: TRADE
-- ══════════════════════════════════════════
local function doTradeSeatLogic()
    log("── [TRADE] Đang chọn ghế...")
    for i = 1, 3 do
        local recvOk, recvWho = checkSeatOccupied(RECV_SEATS[i])
        if recvOk then
            local tradeOk, _ = checkSeatOccupied(TRADE_SEATS[i])
            if not tradeOk then
                local d = randomDelay()
                log(string.format("✅ Ghế nhận %d có người (%s). Ghế trade %d trống. Chờ %.1fs.", i, recvWho, i, d))
                task.wait(d)
                if safeFly(TRADE_SEATS[i]) then
                    currentSeatIdx = i; seatStartTime = tick()
                    log(string.format("🪑 Ngồi Ghế Trade %d. Timeout 120s bắt đầu.", i))
                    return true
                end
                return false
            else
                log(string.format("⏳ Ghế trade %d cũng có người.", i))
            end
        end
    end

    log("⚠ Tất cả ghế nhận trống → tìm ghế trade trống ngẫu nhiên...")
    local empty = {}
    for i = 1,3 do
        if not checkSeatOccupied(TRADE_SEATS[i]) then table.insert(empty, i) end
    end
    if #empty > 0 then
        local c = empty[math.random(1,#empty)]
        local d = randomDelay()
        log(string.format("🎲 Ghế Trade %d. Chờ %.1fs...", c, d))
        task.wait(d)
        if safeFly(TRADE_SEATS[c]) then
            currentSeatIdx = c; seatStartTime = tick()
            log(string.format("🪑 Ngồi Ghế Trade %d. Timeout 120s.", c))
            return true
        end
    else
        log("⛔ Tất cả ghế trade đều có người."); task.wait(3)
    end
    return false
end

-- ══════════════════════════════════════════
--  Timeout Watcher
-- ══════════════════════════════════════════
local function startTimeoutWatcher()
    task.spawn(function()
        while isRunning do
            task.wait(1)
            if seatStartTime and currentSeatIdx then
                local elapsed = tick() - seatStartTime
                if elapsed >= CONFIG.seatTimeout then
                    log(string.format("⏰ TIMEOUT (%.0fs > 120s). Đứng dậy & reset...", elapsed))
                    standUpAndReturn()
                end
            end
        end
    end)
end

-- ══════════════════════════════════════════
--  MAIN LOOP
-- ══════════════════════════════════════════
local function mainLoop()
    log("═══════════════════════════════════════════════")
    log("  AUTO TRADE FRUITS khởi động!")
    log(string.format("  ★ Trái config: %s (dùng chung toàn bộ script)", _G.FruitToTrade))
    log("═══════════════════════════════════════════════")

    character = player.Character or player.CharacterAdded:Wait()
    rootPart  = character:WaitForChild("HumanoidRootPart")
    humanoid  = character:WaitForChild("Humanoid")

    log("📍 Bay về tọa độ chờ ban đầu...")
    pcall(safeFly, POS_WAIT)
    task.wait(1)

    -- Xác định role lần đầu
    determineRole()

    -- Spawn đúng sub-script theo role ngay khi khởi động
    startSubScript()

    -- Khởi động timeout watcher
    startTimeoutWatcher()

    -- ── Main loop ──
    while isRunning do
        pcall(function()
            character = player.Character or player.CharacterAdded:Wait()
            rootPart  = character:WaitForChild("HumanoidRootPart")
            humanoid  = character:WaitForChild("Humanoid")
        end)

        if currentSeatIdx ~= nil then
            -- Đang ngồi ghế → chờ (sub-script đang xử lý trade)
            task.wait(1)
        else
            -- Chưa ngồi ghế → tìm ghế
            if role == "trade" then
                local ok = pcall(doTradeSeatLogic)
                if not ok then
                    log("❌ doTradeSeatLogic lỗi! Reset..."); pcall(safeFly, POS_WAIT); task.wait(2)
                end
            elseif role == "recv" then
                local ok = pcall(doRecvSeatLogic)
                if not ok then
                    log("❌ doRecvSeatLogic lỗi! Reset..."); pcall(safeFly, POS_WAIT); task.wait(2)
                end
            else
                determineRole(); startSubScript(); task.wait(1)
            end

            if currentSeatIdx == nil then
                local rd = randomDelay()
                log(string.format("🔁 Chưa tìm được ghế. Retry sau %.1fs...", rd))
                task.wait(rd)
            end
        end
    end

    log("🛑 Script đã dừng.")
end

-- ══════════════════════════════════════════
--  ENTRY POINT
-- ══════════════════════════════════════════
local success, err = pcall(mainLoop)
if not success then
    warn("[AutoTrade] Lỗi nghiêm trọng: " .. tostring(err))
    warn("[AutoTrade] Re-execute để chạy lại.")
end
    print(">>> SCRIPT FARM ĐANG CHẠY...")
end

    print("------------------------------------------")
    print("KEY: " .. Config.MyKey)
    print("HẠN DÙNG: " .. FormatTime(timeLeft))
    print("SỐ TAB: " .. result .. "/" .. Config.MaxTabs)
    print("------------------------------------------")

    StartFarmScript()
    
    task.spawn(function()
        while task.wait(5) do
            local ok, msg = CheckAuth()
            if not ok then
                game.Players.LocalPlayer:Kick("\n[CẢNH BÁO]\n" .. msg)
                break
            end
        end
    end)
end
