local function GetDB()
    local success, response = pcall(function()
        return request({
            Url = DatabaseURL,
            Method = "GET"
        })
    end)
    if success and response.StatusCode == 200 and response.Body ~= "null" then
        return HttpService:JSONDecode(response.Body)
    end
    return {}
end

local function SaveDB(db)
    pcall(function()
        request({
            Url = DatabaseURL,
            Method = "PUT",
            Headers = {["Content-Type"] = "application/json"},
            Body = HttpService:JSONEncode(db)
        })
    end)
end

local function FormatTime(seconds)
    if seconds == -1 then return "Vĩnh viễn" end
    if seconds <= 0 then return "Đã hết hạn" end
    local days = math.floor(seconds / 86400)
    local hours = math.floor((seconds % 86400) / 3600)
    local mins = math.floor((seconds % 3600) / 60)
    if days > 0 then return string.format("%d ngày, %d giờ", days, hours) end
    return string.format("%d phút, %d giây", mins, seconds % 60)
end

local MyTabID = tostring(math.random(100000, 999999))

local function CheckAuth()
    local inputKey = Config.MyKey
    local currentTime = os.time()
    local db = GetDB()

    -- 1. Check xem key có trong danh sách trắng không
    local keyDuration = Config.DanhSachKey[inputKey]
    if not keyDuration then 
        return false, "Key không tồn tại trong hệ thống!" 
    end

    -- 2. Lấy dữ liệu trên Firebase
    if not db[inputKey] then
        db[inputKey] = {StartTime = 0, IsExpired = false, ActiveTabs = {}}
    end
    local kData = db[inputKey]

    -- 3. Check xem đã bị khóa vĩnh viễn chưa
    if kData.IsExpired then 
        return false, "Key này đã hết hạn sử dụng trước đó!" 
    end

    -- 4. Kích hoạt lần đầu
    if kData.StartTime == 0 then
        kData.StartTime = currentTime
    end

    -- 5. Tính toán thời gian còn lại
    local timeLeft = -1
    if keyDuration ~= -1 then
        local timePassed = currentTime - kData.StartTime
        timeLeft = keyDuration - timePassed
        
        if timePassed >= keyDuration then
            kData.IsExpired = true
            db[inputKey] = kData
            SaveDB(db)
            return false, "Thời hạn key đã hết ("..FormatTime(keyDuration)..")!"
        end
    end

    -- 6. Quản lý số Tab (Concurrency)
    local updatedTabs = {}
    local onlineCount = 0
    for id, lastSeen in pairs(kData.ActiveTabs or {}) do
        if (currentTime - lastSeen) < 15 then
            updatedTabs[id] = lastSeen
            onlineCount = onlineCount + 1
        end
    end

    if not updatedTabs[MyTabID] then
        if onlineCount >= Config.MaxTabs then
            return false, "Key đã đạt giới hạn " .. Config.MaxTabs .. " tab!"
        end
        onlineCount = onlineCount + 1
    end

    updatedTabs[MyTabID] = currentTime
    kData.ActiveTabs = updatedTabs
    db[inputKey] = kData
    
    SaveDB(db)
    return true, onlineCount, timeLeft
end

local function getFruitInfo(serverName)
    local lower = string.lower(serverName)
    for _, f in ipairs(FruitsData) do
        local fname = string.lower(f.name)
        local checks = {fname}
        if ALIASES[fname] then
            for _, a in ipairs(ALIASES[fname]) do table.insert(checks, a) end
        end
        for _, alias in ipairs(checks) do
            if string.find(lower, alias) then return f end
        end
    end
    return nil
end

local function safeFly(targetPos)
    character = player.Character or player.CharacterAdded:Wait()
    rootPart  = character:WaitForChild("HumanoidRootPart")
    enableNoclip()

    local bv = Instance.new("BodyVelocity")
    bv.MaxForce = Vector3.new(math.huge, math.huge, math.huge)
    bv.Velocity = Vector3.new(0,0,0)
    bv.Parent   = rootPart

    local ok = true

    if rootPart.Position.Y < 50 then
        local ascend   = Vector3.new(rootPart.Position.X, CONFIG.safeHeight, rootPart.Position.Z)
        local tweenUp  = TweenService:Create(rootPart,
            TweenInfo.new((rootPart.Position - ascend).Magnitude / CONFIG.speed, Enum.EasingStyle.Linear),
            {CFrame = CFrame.new(ascend)})

        local prev, timer = rootPart.Position, 0
        local conn = RunService.Heartbeat:Connect(function(dt)
            timer += dt
            if timer >= CONFIG.stuckCheckTime then
                if (rootPart.Position - prev).Magnitude < CONFIG.stuckThreshold then
                    ok = false; tweenUp:Cancel()
                end
                timer = 0; prev = rootPart.Position
            end
        end)
        tweenUp:Play(); tweenUp.Completed:Wait(); conn:Disconnect()
    end

    if not ok then
        disableNoclip(); bv:Destroy(); emergencyTeleport(POS_WAIT); return false
    end

    local tweenMain = TweenService:Create(rootPart,
        TweenInfo.new((rootPart.Position - targetPos).Magnitude / CONFIG.speed, Enum.EasingStyle.Linear),
        {CFrame = CFrame.new(targetPos)})

    local prev, timer = rootPart.Position, 0
    local conn = RunService.Heartbeat:Connect(function(dt)
        timer += dt
        if timer >= CONFIG.stuckCheckTime then
            if (rootPart.Position - prev).Magnitude < CONFIG.stuckThreshold then
                ok = false; tweenMain:Cancel()
            end
            timer = 0; prev = rootPart.Position
        end
    end)
    tweenMain:Play(); tweenMain.Completed:Wait(); conn:Disconnect()

    disableNoclip(); bv:Destroy()

    if not ok then
        emergencyTeleport(POS_WAIT); return false
    end
    return true
end

local function normalizeFruitName(raw)
    return (raw:match("^(.-)%-") or raw):lower():gsub("%s+","")
end

local _fruitSet, _cacheTime, CACHE_TTL = nil, 0, 30

local function fetchInventoryFruits()
    local commF = getCommF()
    if not commF then return nil end
    local ok, inv = pcall(function() return commF:InvokeServer("getInventory") end)
    if not ok or not inv then return nil end
    local set = {}
    for _, item in pairs(inv) do
        if item.Type == "Blox Fruit" then
            set[normalizeFruitName(item.Name)] = true
        end
    end
    return set
end

local function invalidateInventoryCache()
    _fruitSet = nil; _cacheTime = 0
end

local function hasFruitInInventory(fruitName)
    local now = tick()
    if _fruitSet == nil or (now - _cacheTime) > CACHE_TTL then
        _fruitSet = fetchInventoryFruits(); _cacheTime = now
    end
    if _fruitSet == nil then return false end
    local target = fruitName:lower():gsub("%s+","")
    return _fruitSet[target] == true
end

-- ══════════════════════════════════════════
--  SUB-SCRIPT: ACC RECV
-- ══════════════════════════════════════════
local recvScriptRunning = false

local function runRecvScript()
    recvScriptRunning = true

    local tradeFunc = getTradeFunction()
    local commF     = getCommF()
    if not tradeFunc or not commF then
        recvScriptRunning = false; return
    end

    local enemyInfo = getFruitInfo(_G.FruitToTrade)
    if not enemyInfo then
        recvScriptRunning = false; return
    end

    local minValue = enemyInfo.beli * 0.6
    local maxValue = math.floor(enemyInfo.beli / 0.6)

    while recvScriptRunning do
        local ok0, inv = pcall(function() return commF:InvokeServer("getInventory") end)
        if not ok0 or not inv then
            task.wait(2); continue
        end

        local myFruits = {}
        for _, item in pairs(inv) do
            if item.Type == "Blox Fruit" then
                local info = getFruitInfo(item.Name)
                if info then
                    local count = item.Count or 1
                    for _ = 1, count do table.insert(myFruits, info) end
                end
            end
        end
        table.sort(myFruits, function(a, b) return a.beli > b.beli end)

        local totalValue, addedCount = 0, 0

        for _, fruit in ipairs(myFruits) do
            if not recvScriptRunning then break end
            if addedCount >= 4 then break end
            if totalValue >= minValue then break end
            if totalValue + fruit.beli > maxValue then
                continue
            end

            local ok1, _ = pcall(function()
                return tradeFunc:InvokeServer("addItem", fruit.id, 1)
            end)
            if ok1 then
                totalValue  += fruit.beli
                addedCount  += 1
            end
            task.wait(0.5)
        end

        if totalValue >= minValue then
            task.wait(10)
            pcall(function() tradeFunc:InvokeServer("accept") end)
        end

        pcall(function()
            if CoreGui:FindFirstChild("Banana Hub") then
                CoreGui["Banana Hub"].Enabled = false
            end
        end)

        if recvScriptRunning then task.wait(1) end
    end
end

end

--  standUpAndReturn
-- ══════════════════════════════════════════
local function standUpAndReturn()
    currentSeatIdx = nil
    seatStartTime  = nil

    stopSubScripts()

    local player = game.Players.LocalPlayer
    local character = player.Character
    if character then
        local humanoid = character:FindFirstChildOfClass("Humanoid")
        if humanoid then
            humanoid.Jump = true
            humanoid.Sit = false
        end
    end

    task.wait(0.5)
    safeFly(POS_WAIT)
    task.wait(1)

    invalidateInventoryCache()
    determineRole()
    startSubScript()
end

-- ══════════════════════════════════════════
--  Seat Logic: TRADE
--  Ngồi theo thứ tự 1→2→3:
--    Ưu tiên ghế có người recv đối diện (cùng index).
--    Nếu không có cặp nào → lấy ghế trade trống đầu tiên theo thứ tự.
--    Recheck sau delay trước khi bay.
-- ══════════════════════════════════════════
local function doTradeSeatLogic()
    -- Bước 1: tìm ghế có người recv đối diện (index nhỏ nhất trước)
    for i = 1, 3 do
        local recvOk, _ = checkSeatOccupied(RECV_SEATS[i])
        if recvOk then
            local tradeOk, _ = checkSeatOccupied(TRADE_SEATS[i])
            if not tradeOk then
                local d = randomDelay()
                task.wait(d)
                -- Recheck sau delay
                local recvStill, _ = checkSeatOccupied(RECV_SEATS[i])
                local tradeStill, _ = checkSeatOccupied(TRADE_SEATS[i])
                if not recvStill or tradeStill then return false end
                if safeFly(TRADE_SEATS[i]) then
                    currentSeatIdx = i; seatStartTime = tick()
                    return true
                end
                return false
            end
        end
    end

    -- Bước 2: không có cặp nào → ngồi ghế trade trống đầu tiên theo thứ tự 1→2→3
    for i = 1, 3 do
        local tradeOk, _ = checkSeatOccupied(TRADE_SEATS[i])
        if not tradeOk then
            local d = randomDelay()
            task.wait(d)
            -- Recheck sau delay
            local tradeStill, _ = checkSeatOccupied(TRADE_SEATS[i])
            if tradeStill then return false end
            if safeFly(TRADE_SEATS[i]) then
                currentSeatIdx = i; seatStartTime = tick()
                return true
            end
            return false
        end
    end

    -- Tất cả ghế trade đều có người
    task.wait(3)
    return false
end

-- ══════════════════════════════════════════
--  Timeout Watcher
-- ══════════════════════════════════════════
local function startTimeoutWatcher()
    task.spawn(function()
        while isRunning do
            task.wait(1)
            if seatStartTime and currentSeatIdx then
                local elapsed = tick() - seatStartTime
                if elapsed >= CONFIG.seatTimeout then
                    standUpAndReturn()
                end
            end
        end
    end)
end

-- ══════════════════════════════════════════
--  MAIN LOOP
-- ══════════════════════════════════════════
local function mainLoop()
    character = player.Character or player.CharacterAdded:Wait()
    rootPart  = character:WaitForChild("HumanoidRootPart")
    humanoid  = character:WaitForChild("Humanoid")

    pcall(safeFly, POS_WAIT)
    task.wait(1)

    determineRole()
    startSubScript()
    startTimeoutWatcher()

    while isRunning do
        pcall(function()
            character = player.Character or player.CharacterAdded:Wait()
            rootPart  = character:WaitForChild("HumanoidRootPart")
            humanoid  = character:WaitForChild("Humanoid")
        end)

        if currentSeatIdx ~= nil then
            task.wait(1)
        else
            if role == "trade" then
                local ok = pcall(doTradeSeatLogic)
                if not ok then
                    pcall(safeFly, POS_WAIT); task.wait(2)
                end
            elseif role == "recv" then
                local ok = pcall(doRecvSeatLogic)
                if not ok then
                    pcall(safeFly, POS_WAIT); task.wait(2)
                end
            else
                determineRole(); startSubScript(); task.wait(1)
            end

            if currentSeatIdx == nil then
                local rd = randomDelay()
                task.wait(rd)
            end
        end
    end
end

-- ══════════════════════════════════════════
--  ENTRY POINT
-- ══════════════════════════════════════════
local success, err = pcall(mainLoop)
if not success then
    warn("[AutoTrade] Lỗi nghiêm trọng: " .. tostring(err))
    warn("[AutoTrade] Re-execute để chạy lại.")
end
    print("...")
end

if not success then
    game.Players.LocalPlayer:Kick("\n[HỆ THỐNG KEY ONLINE]\n" .. result)
else
    print("------------------------------------------")
    print("KEY: " .. Config.MyKey)
    print("HẠN DÙNG: " .. FormatTime(timeLeft))
    print("------------------------------------------")

    StartFarmScript()
    
    task.spawn(function()
        while task.wait(5) do
            local ok, msg = CheckAuth()
            if not ok then
                game.Players.LocalPlayer:Kick("\n[CẢNH BÁO]\n" .. msg)
                break
            end
        end
    end)
end
